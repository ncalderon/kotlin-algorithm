3    import java.util.AbstractList;
4    import java.util.List;
5    import java.util.stream.Collectors;

7    public class IovlabsChallenge {
8    
9        private static List<Character> convertStringToCharList(final String s) {
10           return new AbstractList<Character>() {
11               @Override
12               public Character get(int index) {
13                   return s.charAt(index);
14               }
15   
16               @Override
17               public int size() {
18                   return s.length();
19               }
20           };
21       }
22   
23       public static String solution(String s) {
24           List<Character> p = convertStringToCharList(s);
25           for (int i = 0; i < s.length(); i++) {
26               Character c = p.get(i);
27               Character ic = p.get(s.length()-i);
28               if (c != '?' && ic != '?' && c != ic){
29                   return "NO";
30               }
31   
32               if (c == '?' && ic == '?'){
33                   p.set(i, 'a');
34                   p.set(s.length()-i, 'a');
35               }
36   
37               if (c != '?' && ic == '?'){
38                   p.set(s.length()-i, c);
39               }
40   
41               if (c == '?' && ic != '?'){
42                   p.set(i, ic);
43               }
44           }
45           return p.stream()
46                   .map(String::valueOf)
47                   .collect(Collectors.joining());
48       }
49   
50       public static void main (String[]args){
51           assert solution("?ab??a").equals("aabbaa");
52           assert solution("bab??a").equals("NO");
53           assert solution("?a?").equals("aaa");
54           assert solution("radar").equals("radar");
55           assert solution("mom").equals("mom");
56           assert solution("a?n?").equals("anna");
57           assert solution("a?").equals("aa");
58           assert solution("a").equals("a");
59   
60       }
